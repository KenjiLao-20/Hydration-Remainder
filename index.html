Hydration Reminder

Stay hydrated!

from pyscript import Element import time import threading def remind(interval): while True: time.sleep(interval * 60) # Convert minutes to seconds Element("message").element.innerText = "Time to drink water! ðŸ’§" def start_reminder(event): interval = int(Element("interval").element.value) if interval > 0: threading.Thread(target=remind, args=(interval,), daemon=True).start() Element("start").element.disabled = True Element("stop").element.disabled = False def stop_reminder(event): Element("message").element.innerText = "Stay hydrated!" Element("start").element.disabled = False Element("stop").element.disabled = True Element("start").element.bind("click", start_reminder) Element("stop").element.bind("click", stop_reminder)
