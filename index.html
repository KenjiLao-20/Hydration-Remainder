from pyscript import Element def reset_result(): """ Function to reset the result display and error message.""" Element("result").element.style.display = "none" # Hide the result section Element("error").element.innerHTML = "" # Clear error message Element("waterIntake").element.innerHTML = "" def calculate_hydration(event): event.preventDefault() # Prevent the form from refreshing the page reset_result() # Reset any previous result when recalculating try: weight = float(Element("weight").element.value) activity_level = Element("activityLevel").element.value except ValueError: Element("error").element.innerHTML = "Please enter a valid input." return # Base required water intake based on weight base_required_water = weight * 0.5 # 0.5 oz per pound # Additional water intake based on activity level if activity_level == "none": required_water = base_required_water # No additional water needed elif activity_level == "low": hydration_time = 30 # 30 minutes additional_water = (hydration_time / 30) * 12 # 12 oz for every 30 minutes required_water = base_required_water + additional_water elif activity_level == "moderate": hydration_time = 60 # 1 hour additional_water = (hydration_time / 30) * 12 required_water = base_required_water + additional_water elif activity_level == "high": hydration_time = 90 # 1 hour & 30 minutes additional_water = (hydration_time / 30) * 12 required_water = base_required_water + additional_water else: Element("error").element.innerHTML = "Please select a valid activity level." return Element("waterIntake").element.innerHTML = f"{required_water:.2f}" # Show result section Element("result").element.style.display = "block" Element("calculateBtn").element.onclick = calculate_hydration
