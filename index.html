from pyscript import Element, window def start_reminder(event): event.preventDefault() try: interval = int(Element("interval").element.value) except ValueError: Element("error").element.innerHTML = "Please enter a valid interval." return if interval < 1: Element("error").element.innerHTML = "Interval must be at least 1 minute." return # Start the hydration reminder Element("result").element.innerHTML = f"Hydration reminder set for every {interval} minutes." Element("result").element.style.display = "block" # Show notification pop-up window.alert("Drink Some Water! Stay Hydrated!") Element("startReminder").element.onclick = start_reminder
